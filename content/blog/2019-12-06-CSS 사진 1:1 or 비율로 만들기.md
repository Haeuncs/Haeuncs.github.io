```JavaScript
const HashTagImageWrapperOuter = styled.div`
  position: relative;
  width: 100%;

    content: '';
    display: block;
    padding-top: 140%;
  }
`;

  position: absolute;
  top: 0;
  right: 0;
  bottom: 0;
  left: 0;
`;
```
